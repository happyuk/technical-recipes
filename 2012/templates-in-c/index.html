Now suppose we wish to do this in a generic way, allowing us to store and print not only integers, but any kind of primitive data type.  The equivalent template class for class A would now look like this:


#include <iostream>

template< class T>
class A
{
private:
    T data;

public:
    A( const T& d );
    T& Value() const;
    void Print() const;
};

template<class T>
A<T>::A( const T& d )
{
    data = d;
}

template<class T>
T& A<T>::Value() const
{
    return data;
